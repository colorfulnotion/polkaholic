```
SELECT date(block_time) as logDT, MIN(number) startBN, MAX(number) endBN, COUNT(*) numBlocks FROM `substrate-etl.kusama.block0`  where signed and LAST_DAY(date(block_time)) = "2022-05-31" group by logDT order by logDT
```
